## Zyfra - Datapicker Multi

<Subtitle>Datepicker Multi supports 'absolute' and 'relative' mode date.</Subtitle>

Import module:

```typescript
import { ZyfraDatepickerModule } from '@digital-plant/zyfra-components';

@NgModule({
  imports: [ZyfraDatepickerModule],
})
export class YourModule {}
```

To component:

```html
<zyfra-datepicker-multi
  [mode]="absolute"
  [label]="'Время'"
  [placeholder]="'Введите значение'"
  [disabled]="false"
  [showClear]="true"
  [relativeValue]="''"
  [ngModel]="'21.09.2021 20:17:07'"
  [dateFormat]="'DD.MM.YYYY'"
  [timeFormat]="'HH:mm:ss'"
  [returnFormatValue]="'string'"
  [required]="true"
  [spanClass]="null"
  [locale]="'ru_RU'"
  [showDate]="false"
  [showTime]="true"
  [monthNavigator]="true"
  [yearNavigator]="false"
  [yearRange]="false"
  [showCalendarBtn]="null"
  [calendarBtnLabel]="true"
  [numberOfMonths]="'Выбрать'"
  [selectionMode]="false"
  [disabledDates]="1"
  [firstDayOfWeek]="'single'"
>
</zyfra-datepicker-multi>
```
