## Zyfra - Password

<Subtitle>Password displays strength indicator for password fields</Subtitle>

Import module:

```typescript
import { ZyfraPasswordModule } from '@digital-plant/zyfra-components';

@NgModule({
  imports: [ZyfraPasswordModule],
})
export class YourModule {}
```

To component:

```html
<zyfra-password
  [(ngModel)]="passwordValue"
  [toggleMask]="true"
  [required]="true"
  [disabled]="false"
  [mediumRegex]="mediumRegex"
  [strongRegex]="strongRegex"
  weakLabel="Слабый пароль"
  mediumLabel="Средний пароль"
  strongLabel="Сильный пароль"
  label="Пароль"
  promptLabel="Введите пароль">
</zyfra-password>

```

```typescript
public passwordValue = '';

/*
 * Default mediumRegex, matches the following rules:
 *
 * At least one lowercase
 * At least one uppercase or one number
 * Minimum 6 characters
 */
public mediumRegex = '^(((?=.*[a-z])(?=.*[A-Z]))|((?=.*[a-z])(?=.*[0-9]))|((?=.*[A-Z])(?=.*[0-9])))(?=.{6,}).';

/*
 * Default strongRegex, matches the following rules:
 *
 * At least one lowercase
 * At least one uppercase
 * At least one numeric
 * Minimum 8 characters
 */
public strongRegex = '^(?=.*[a-z])(?=.*[A-Z])(?=.*[0-9])(?=.{8,})';

```
