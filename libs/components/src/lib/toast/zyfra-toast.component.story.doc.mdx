## Zyfra - Toast

<Subtitle>Toast is used to display messages in an overlay</Subtitle>

Import module:

```typescript
import { ZyfraToastModule } from '@digital-plant/zyfra-components';
import { ZyfraButtonModule } from '@digital-plant/zyfra-components';

@NgModule({
  imports: [ZyfraToastModule, ZyfraButtonModule],
})
export class YourModule {}
```

Component:

```ts
@Component({
  selector: 'toast-container',
  template: `
    <zyfra-button (click)="showMessage('myKey0')" label="show custom"></zyfra-button>
    <zyfra-button (click)="showMessage()" label="show default"></zyfra-button>

    <!--  Default toast  -->
    <zyfra-toast></zyfra-toast>

    <!--  Custom toast  -->
    <zyfra-toast
      [key]="'myKey0'"
      [position]="position"
      [showTransitionOptions]="showTransitionOptions"
      [preventOpenDuplicates]="preventOpenDuplicates"
      [preventDuplicates]="preventDuplicates"
      [customTemplate]="useCustomTemplate && customTemplate"
    ></zyfra-toast>

    <ng-template #customTemplate>
      <div>
        <p>content</p>
        <ul>
          <li>item 1 (for example)</li>
          <li><b>item 2</b></li>
        </ul>
      </div>
    </ng-template>
  `,
  providers: [ToastService],
})
class MyToastContainerComponent {
  @Input() key;
  @Input() position;
  @Input() showTransitionOptions;
  @Input() preventOpenDuplicates;
  @Input() preventDuplicates;
  @Input() useCustomTemplate;

  constructor(private toast: ToastService) {}

  public showMessage(key?: string) {
    this.toast.add({ key, severity: 'success', summary: 'Summary Text', detail: 'Detail Text' });
  }
}
```
